function fibonacci(n) {
  if (n <= 0) return null;
  if (n == 1 || n == 2) return 1;
  
  let a = 1, b = 1;
  
  for (let i = 2; i < n; i++) {
    let temp = a;
    a += b;
    b = temp;
    }
    
  return a;
};

console.log(fibonacci(15));
